<body>
  <h2>Smart Water Management Dashboard</h2>

  <canvas id="usageChart" width="600" height="300"></canvas>

  <h3>Water Usage Data</h3>
  <table id="dataTable">
    <thead>
      <tr>
        <th>Date</th>
        <th>Usage (Liters)</th>
        <th>Leak Detected</th>
      </tr>
    </thead>
    <tbody></tbody>
  </table>

  <script>
    async function loadData() {
      const response = await fetch('/api/data');
      const data = await response.json();

      // Fill table
      let tableBody = document.querySelector("#dataTable tbody");
      tableBody.innerHTML = "";
      data.forEach(row => {
        let tr = `<tr>
          <td>${row.Date}</td>
          <td>${row.Usage}</td>
          <td>${row.Leak_Detected}</td>
        </tr>`;
        tableBody.innerHTML += tr;
      });

      // Draw chart
      const ctx = document.getElementById('usageChart').getContext('2d');
      new Chart(ctx, {
        type: 'line',
        data: {
          labels: data.map(r => r.Date),
          datasets: [{
            label: 'Water Usage (Liters)',
            data: data.map(r => r.Usage),
            borderColor: 'blue',
            fill: false
          }]
        }
      });
    }
    loadData();
  </script>
